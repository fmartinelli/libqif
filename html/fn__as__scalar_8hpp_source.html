59 template<typename T1, typename T2>
60 inline
61 typename T1::elem_type
62 as_scalar_redirect<2>::apply(const Glue<T1, T2, glue_times>& X)
63  {
64  arma_extra_debug_sigprint();
65 
66  typedef typename T1::elem_type eT;
67 
68  // T1 must result in a matrix with one row
69  // T2 must result in a matrix with one column
70 
71  const bool has_all_mat = (is_Mat<T1>::value || is_Mat_trans<T1>::value) && (is_Mat<T2>::value || is_Mat_trans<T2>::value);
72  const bool prefer_at_accessor = Proxy<T1>::prefer_at_accessor || Proxy<T2>::prefer_at_accessor;
73 
74  const bool do_partial_unwrap = has_all_mat || prefer_at_accessor;
75 
76  if(do_partial_unwrap == true)
77  {
78  const partial_unwrap<T1> tmp1(X.A);
79  const partial_unwrap<T2> tmp2(X.B);
80 
81  typedef typename partial_unwrap<T1>::stored_type TA;
82  typedef typename partial_unwrap<T2>::stored_type TB;
83 
84  const TA& A = tmp1.M;
85  const TB& B = tmp2.M;
86 
87  const uword A_n_rows = (tmp1.do_trans == false) ? (TA::is_row ? 1 : A.n_rows) : (TA::is_col ? 1 : A.n_cols);
88  const uword A_n_cols = (tmp1.do_trans == false) ? (TA::is_col ? 1 : A.n_cols) : (TA::is_row ? 1 : A.n_rows);
89 
90  const uword B_n_rows = (tmp2.do_trans == false) ? (TB::is_row ? 1 : B.n_rows) : (TB::is_col ? 1 : B.n_cols);
91  const uword B_n_cols = (tmp2.do_trans == false) ? (TB::is_col ? 1 : B.n_cols) : (TB::is_row ? 1 : B.n_rows);
92 
93  arma_debug_check( (A_n_rows != 1) || (B_n_cols != 1) || (A_n_cols != B_n_rows), "as_scalar(): incompatible dimensions" );
94 
95  const eT val = op_dot::direct_dot(A.n_elem, A.memptr(), B.memptr());
96 
97  return (tmp1.do_times || tmp2.do_times) ? (val * tmp1.get_val() * tmp2.get_val()) : val;
98  }
99  else
100  {
101  const Proxy<T1> PA(X.A);
102  const Proxy<T2> PB(X.B);
103 
104  arma_debug_check
105  (
106  (PA.get_n_rows() != 1) || (PB.get_n_cols() != 1) || (PA.get_n_cols() != PB.get_n_rows()),
107  "as_scalar(): incompatible dimensions"
108  );
109 
110  return op_dot::apply_proxy(PA,PB);
111  }
112  }

119 as_scalar_redirect<3>::apply(const Glue< Glue<T1, T2, glue_times>, T3, glue_times >& X)

124 
125  // T1 * T2 must result in a matrix with one row
126  // T3 must result in a matrix with one column

150

152  const Mat<eT>& B = tmp2.M;
153  const Mat<eT>& C = tmp3.M;
154 
155  const uword A_n_rows = (tmp1.do_trans == false) ? A.n_rows : A.n_cols;
156  const uword A_n_cols = (tmp1.do_trans == false) ? A.n_cols : A.n_rows;

159

174  ,
