217 template<typename T1>
218 inline
219 void
220 op_all::apply_helper(Mat<uword>& out, const Proxy<T1>& P, const uword dim)
221  {
222  arma_extra_debug_sigprint();
223 
224  const uword n_rows = P.get_n_rows();
225  const uword n_cols = P.get_n_cols();
226 
227  typedef typename Proxy<T1>::elem_type eT;
228 
229  if(dim == 0) // traverse rows (ie. process each column)
230  {
231  out.zeros(1, n_cols);
232 
233  if(out.n_elem == 0) { return; }
234 
235  uword* out_mem = out.memptr();
236 
237  if(is_Mat<typename Proxy<T1>::stored_type>::value == true)
238  {
239  const unwrap<typename Proxy<T1>::stored_type> U(P.Q);
240 
241  for(uword col=0; col < n_cols; ++col)
242  {
243  const eT* colmem = U.M.colptr(col);
244 
245  uword count = 0;
246 
247  for(uword row=0; row < n_rows; ++row)
248  {
249  if(colmem[row] != eT(0)) { ++count; }
250  }
251 
252  out_mem[col] = (n_rows == count) ? uword(1) : uword(0);
253  }
254  }
255  else
256  {
257  for(uword col=0; col < n_cols; ++col)
258  {
259  uword count = 0;
260 
261  for(uword row=0; row < n_rows; ++row)
262  {
263  if(P.at(row,col) != eT(0)) { ++count; }
264  }
265 
266  out_mem[col] = (n_rows == count) ? uword(1) : uword(0);
267  }
268  }
269  }
270  else
271  {
272  out.zeros(n_rows, 1);
273 
274  uword* out_mem = out.memptr();
275 
276  // internal dual use of 'out': keep the counts for each row
277 
278  if(is_Mat<typename Proxy<T1>::stored_type>::value == true)
279  {
280  const unwrap<typename Proxy<T1>::stored_type> U(P.Q);
281 
282  for(uword col=0; col < n_cols; ++col)
283  {
284  const eT* colmem = U.M.colptr(col);
285 
286  for(uword row=0; row < n_rows; ++row)
287  {
288  if(colmem[row] != eT(0)) { ++out_mem[row]; }
289  }
290  }
291  }
292  else
293  {
294  for(uword col=0; col < n_cols; ++col)
295  {
296  for(uword row=0; row < n_rows; ++row)
297  {
298  if(P.at(row,col) != eT(0)) { ++out_mem[row]; }
299  }
300  }
301  }
302 
303 
304  // see what the counts tell us
305 
306  for(uword row=0; row < n_rows; ++row)
307  {
308  out_mem[row] = (n_cols == out_mem[row]) ? uword(1) : uword(0);
309  }
310 
311  }
312  }
