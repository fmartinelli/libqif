685  {

688  const uword t_n_cols = Mat<eT>::n_cols;
689 
690  const uword A_n_cols = col_num;
691  const uword B_n_cols = t_n_cols - col_num;
692 
693  // insertion at col_num == n_cols is in effect an append operation

700  eT* out_mem = out.memptr();

702 
703  if(A_n_cols > 0)

706  }
707 
708  if(B_n_cols > 0)
