27 /***********************************************************************
28 * Sparse Vector Area (SVA) is a container for sparse vectors. This
29 * program object is used mainly on computing factorization, where the
30 * sparse vectors are rows and columns of sparse matrices.
31 *
32 * The SVA storage is a set of locations numbered 1, 2, ..., size,
33 * where size is the size of SVA, which is the total number of
34 * locations currently allocated. Each location is identified by its
35 * pointer p, 1 <= p <= size, and is the pair (ind[p], val[p]), where
36 * ind[p] and val[p] are, respectively, the index and value fields used
37 * to store the index and numeric value of a particular vector element.
38 *
39 * Each sparse vector is identified by its reference number k,
40 * 1 <= k <= n, where n is the total number of vectors currently stored
41 * in SVA, and defined by the triplet (ptr[k], len[k], cap[k]), where:
42 * ptr[k] is a pointer to the first location of the vector; len[k] is
43 * the vector length, which is the number of its non-zero elements,
44 * len[k] >= 0; and cap[k] is the capacity of the vector, which is the
45 * total number of adjacent locations allocated to that vector,
46 * cap[k] >= len[k]. Thus, non-zero elements of k-th vector are stored
47 * in locations ptr[k], ptr[k]+1, ..., ptr[k]+len[k]-1, and locations
48 * ptr[k]+len[k], ptr[k]+len[k]+1, ..., ptr[k]+cap[k]-1 are reserved.
49 *
50 * The SVA storage is divided into three parts as follows:
51 *
52 * Locations 1, 2, ..., m_ptr-1 constitute the left (dynamic) part of
53 * SVA. This part is used to store vectors, whose capacity may change.
54 * Note that all vectors stored in the left part are also included in
55 * a doubly linked list, where they are ordered by increasing their
56 * pointers ptr[k] (this list is needed for efficient implementation
57 * of the garbage collector used to defragment the left part of SVA);
58 *
59 * Locations m_ptr, m_ptr+1, ..., r_ptr-1 are free and constitute the
60 * middle (free) part of SVA.
61 *
62 * Locations r_ptr, r_ptr+1, ..., size constitute the right (static)
63 * part of SVA. This part is used to store vectors, whose capacity is
64 * not changed. */

68 struct SVA
69 { /* sparse vector area */
70  int n_max;
71  /* maximal value of n (enlarged automatically) */
72  int n;
73  /* number of currently allocated vectors, 0 <= n <= n_max */
74  int *ptr; /* int ptr[1+n_max]; */
75  /* ptr[0] is not used;
76  * ptr[k], 1 <= i <= n, is pointer to first location of k-th
77  * vector in the arrays ind and val */
78  int *len; /* int len[1+n_max]; */
79  /* len[0] is not used;
80  * len[k], 1 <= k <= n, is length of k-th vector, len[k] >= 0 */
81  int *cap; /* int cap[1+n_max]; */
82  /* cap[0] is not used;
83  * cap[k], 1 <= k <= n, is capacity of k-th vector (the number
84  * of adjacent locations allocated to it), cap[k] >= len[k] */
85  /* NOTE: if cap[k] = 0, then ptr[k] = 0 and len[k] = 0 */
86  int size;
87  /* total number of locations in SVA */
88  int m_ptr, r_ptr;
89  /* partitioning pointers that define the left, middle, and right
90  * parts of SVA (see above); 1 <= m_ptr <= r_ptr <= size+1 */
91  int head;
92  /* number of first (leftmost) vector in the linked list */
93  int tail;
94  /* number of last (rightmost) vector in the linked list */
95  int *prev; /* int prev[1+n_max]; */
96  /* prev[0] is not used;
97  * prev[k] is number of vector which precedes k-th vector in the
98  * linked list;
99  * prev[k] < 0 means that k-th vector is not in the list */
100  int *next; /* int next[1+n_max]; */
101  /* next[0] is not used;
102  * next[k] is number of vector which succedes k-th vector in the
103  * linked list;
104  * next[k] < 0 means that k-th vector is not in the list */
105  /* NOTE: only vectors having non-zero capacity and stored in the
106  * left part of SVA are included in this linked list */
107  int *ind; /* int ind[1+size]; */
108  /* ind[0] is not used;
109  * ind[p], 1 <= p <= size, is index field of location p */
110  double *val; /* double val[1+size]; */
111  /* val[0] is not used;
112  * val[p], 1 <= p <= size, is value field of location p */
113 #if 1
114  int talky;
115  /* option to enable talky mode */
116 #endif
117 };
