29 /***********************************************************************
30 * The structure CFG describes the conflict graph.
31 *
32 * Conflict graph is an undirected graph G = (V, E), where V is a set
33 * of vertices, E <= V x V is a set of edges. Each vertex v in V of the
34 * conflict graph corresponds to a binary variable z[v], which is
35 * either an original binary variable x[j] or its complement 1 - x[j].
36 * Edge (v,w) in E means that z[v] and z[w] cannot take the value 1 at
37 * the same time, i.e. it defines an inequality z[v] + z[w] <= 1, which
38 * is assumed to be valid for original MIP.
39 *
40 * Since the conflict graph may be dense, it is stored as an union of
41 * its cliques rather than explicitly. */
42

47 struct CFG
48 { /* conflict graph descriptor */
49  int n;
50  /* number of *all* variables (columns) in corresponding MIP */
51  int *pos; /* int pos[1+n]; */
52  /* pos[0] is not used;
53  * pos[j] = v, 1 <= j <= n, means that vertex v corresponds to
54  * original binary variable x[j], and pos[j] = 0 means that the
55  * conflict graph has no such vertex */
56  int *neg; /* int neg[1+n]; */
57  /* neg[0] is not used;
58  * neg[j] = v, 1 <= j <= n, means that vertex v corresponds to
59  * complement of original binary variable x[j], and neg[j] = 0
60  * means that the conflict graph has no such vertex */
61  DMP *pool;
62  /* memory pool to allocate elements of the conflict graph */
63  int nv_max;
64  /* maximal number of vertices in the conflict graph */
65  int nv;
66  /* current number of vertices in the conflict graph */
67  int *ref; /* int ref[1+nv_max]; */
68  /* ref[v] = j, 1 <= v <= nv, means that vertex v corresponds
69  * either to original binary variable x[j] or to its complement,
70  * i.e. either pos[j] = v or neg[j] = v */
71  CFGVLE **vptr; /* CFGVLE *vptr[1+nv_max]; */
72  /* vptr[v], 1 <= v <= nv, is an initial pointer to the list of
73  * vertices adjacent to vertex v */
74  CFGCLE **cptr; /* CFGCLE *cptr[1+nv_max]; */
75  /* cptr[v], 1 <= v <= nv, is an initial pointer to the list of
76  * cliques that contain vertex v */
77 };
