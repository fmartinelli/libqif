34  /* LU-factorization being computed */
35  /*--------------------------------------------------------------*/
36  /* to efficiently choose pivot elements according to Markowitz
37  * strategy, the search technique proposed by Iain Duff is used;
38  * it is based on using two families of sets {R[0], ..., R[n]}
39  * and {C[0], ..., C[n]}, where R[k] and C[k], 0 <= k <= n, are,
40  * respectively, sets of rows and columns of the active submatrix
41  * of matrix V having k non-zeros (i.e. whose length is k); each
42  * set R[k] and C[k] is implemented as a doubly linked list */
43  int *rs_head; /* int rs_head[1+n]; */
44  /* rs_head[k], 0 <= k <= n, is the number of first row, which
45  * has k non-zeros in the active submatrix */
46  int *rs_prev; /* int rs_prev[1+n]; */
47  /* rs_prev[0] is not used;
48  * rs_prev[i], 1 <= i <= n, is the number of previous row, which
49  * has the same number of non-zeros as i-th row;
50  * rs_prev[i] < 0 means that i-th row is inactive */
51  int *rs_next; /* int rs_next[1+n]; */
52  /* rs_next[0] is not used;
53  * rs_next[i], 1 <= i <= n, is the number of next row, which has
54  * the same number of non-zeros as i-th row;
55  * rs_next[i] < 0 means that i-th row is inactive */
56  int *cs_head; /* int cs_head[1+n]; */
57  /* cs_head[k], 0 <= k <= n, is the number of first column, which
58  * has k non-zeros in the active submatrix */
59  int *cs_prev; /* int cs_prev[1+n]; */
60  /* cs_prev[0] is not used;
61  * cs_prev[j], 1 <= j <= n, is the number of previous column,
62  * which has the same number of non-zeros as j-th column;
63  * cs_prev[j] < 0 means that j-th column is inactive */
64  int *cs_next; /* int cs_next[1+n]; */
65  /* cs_next[0] is not used;
66  * cs_next[j], 1 <= j <= n, is the number of next column, which
67  * has the same number of non-zeros as j-th column;
68  * cs_next[j] < 0 means that j-th column is inactive */
69  /* NOTE: cs_prev[j] = cs_next[j] = j means that j-th column was
70  * temporarily removed from corresponding set C[k] by the
71  * pivoting routine according to Uwe Suhl's heuristic */
72  /*--------------------------------------------------------------*/
73  /* working arrays */

75  /* vr_max[0] is not used;
76  * vr_max[i], 1 <= i <= n, is used only if i-th row of matrix V

90  double piv_tol;

121 #define sgf_activate_row(i) \
122  do \
123  { int len = vr_len[i]; \
124  rs_prev[i] = 0; \
125  rs_next[i] = rs_head[len]; \
126  if (rs_next[i] != 0) \
127  rs_prev[rs_next[i]] = i; \
128  rs_head[len] = i; \
129  } while (0)
130 /* include i-th row of matrix V in active set R[len] */

132 #define sgf_deactivate_row(i) \
133  do \
134  { if (rs_prev[i] == 0) \
135  rs_head[vr_len[i]] = rs_next[i]; \
136  else \
137  rs_next[rs_prev[i]] = rs_next[i]; \
138  if (rs_next[i] == 0) \
139  ; \
140  else \
141  rs_prev[rs_next[i]] = rs_prev[i]; \
142  rs_prev[i] = rs_next[i] = -1; \
143  } while (0)
