144 typedef struct
145 { /* this structure contains results reported by the routines which
146  checks Karush-Kuhn-Tucker conditions (for details see comments
147  to those routines) */
148  /*--------------------------------------------------------------*/
149  /* xR - A * xS = 0 (KKT.PE) */
150  double pe_ae_max;
151  /* largest absolute error */
152  int pe_ae_row;
153  /* number of row with largest absolute error */
154  double pe_re_max;
155  /* largest relative error */
156  int pe_re_row;
157  /* number of row with largest relative error */
158  int pe_quality;
159  /* quality of primal solution:
160  'H' - high
161  'M' - medium
162  'L' - low
163  '?' - primal solution is wrong */
164  /*--------------------------------------------------------------*/
165  /* l[k] <= x[k] <= u[k] (KKT.PB) */
166  double pb_ae_max;
167  /* largest absolute error */
168  int pb_ae_ind;
169  /* number of variable with largest absolute error */
170  double pb_re_max;
171  /* largest relative error */
172  int pb_re_ind;
173  /* number of variable with largest relative error */
174  int pb_quality;
175  /* quality of primal feasibility:
176  'H' - high
177  'M' - medium
178  'L' - low
179  '?' - primal solution is infeasible */
180  /*--------------------------------------------------------------*/
181  /* A' * (dR - cR) + (dS - cS) = 0 (KKT.DE) */
182  double de_ae_max;
183  /* largest absolute error */
184  int de_ae_col;
185  /* number of column with largest absolute error */
186  double de_re_max;
187  /* largest relative error */
188  int de_re_col;
189  /* number of column with largest relative error */
190  int de_quality;
191  /* quality of dual solution:
192  'H' - high
193  'M' - medium
194  'L' - low
195  '?' - dual solution is wrong */
196  /*--------------------------------------------------------------*/
197  /* d[k] >= 0 or d[k] <= 0 (KKT.DB) */
198  double db_ae_max;
199  /* largest absolute error */
200  int db_ae_ind;
201  /* number of variable with largest absolute error */
202  double db_re_max;
203  /* largest relative error */
204  int db_re_ind;
205  /* number of variable with largest relative error */
206  int db_quality;
207  /* quality of dual feasibility:
208  'H' - high
209  'M' - medium
210  'L' - low
211  '?' - dual solution is infeasible */
212  /*--------------------------------------------------------------*/
213  /* (x[k] - bound of x[k]) * d[k] = 0 (KKT.CS) */
214  double cs_ae_max;
215  /* largest absolute error */
216  int cs_ae_ind;
217  /* number of variable with largest absolute error */
218  double cs_re_max;
219  /* largest relative error */
220  int cs_re_ind;
221  /* number of variable with largest relative error */
222  int cs_quality;
223  /* quality of complementary slackness:
224  'H' - high
225  'M' - medium
226  'L' - low
227  '?' - primal and dual solutions are not complementary */
228 } LPXKKT;
